import java.util.Scanner;

class Node {
    int data;
    Node next;

    Node(int data) {
        this.data = data;
        this.next = null;
    }
}

class LinkedList {

    Node head;

    // Insert node at end
    void insert(int data) {

        Node newNode = new Node(data);

        if (head == null) {
            head = newNode;
            return;
        }

        Node temp = head;

        while (temp.next != null) {
            temp = temp.next;
        }

        temp.next = newNode;
    }

    // Count positive and negative numbers
    void countPosNeg() {

        int positive = 0;
        int negative = 0;

        Node temp = head;

        while (temp != null) {

            if (temp.data > 0) {
                positive++;
            } 
            else if (temp.data < 0) {
                negative++;
            }

            temp = temp.next;
        }

        System.out.println("Positive = " + positive);
        System.out.println("Negative = " + negative);
    }
}

public class Main {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        LinkedList list = new LinkedList();

        int n = sc.nextInt();

        for (int i = 0; i < n; i++) {
            list.insert(sc.nextInt());
        }

        list.countPosNeg();
    }
}




Input
6
10 -5 20 -3 15 -7

Output
Positive = 3
Negative = 3
