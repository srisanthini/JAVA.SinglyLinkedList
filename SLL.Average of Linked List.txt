import java.util.Scanner;

class Node {
    int data;
    Node next;

    Node(int data) {
        this.data = data;
        this.next = null;
    }
}

class LinkedList {

    Node head;

    // Insert node at end
    void insert(int data) {

        Node newNode = new Node(data);

        if (head == null) {
            head = newNode;
            return;
        }

        Node temp = head;

        while (temp.next != null) {
            temp = temp.next;
        }

        temp.next = newNode;
    }

    // Find average of nodes
    void findAverage() {

        int sum = 0;
        int count = 0;

        Node temp = head;

        while (temp != null) {
            sum = sum + temp.data;
            count++;
            temp = temp.next;
        }

        int avg = sum / count;

        System.out.println("Average = " + avg);
    }
}

public class Main {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        LinkedList list = new LinkedList();

        int n = sc.nextInt();

        for (int i = 0; i < n; i++) {
            list.insert(sc.nextInt());
        }

        list.findAverage();
    }
}


Input
5
10 20 30 40 50
Output
Average = 30
